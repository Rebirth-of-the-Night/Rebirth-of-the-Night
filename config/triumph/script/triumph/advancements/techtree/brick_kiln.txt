setIcon(<pyrotech:stone_kiln>)

addParent("triumph:advancements/techtree/pit_kiln")


//Set the toast notification in the top right corner of the screen to not show up when this advancement is achieved.

//Automatically unlocked if skipped with further advancements
criteria = addCriteria("Refractory Kiln", "triumph:completed_advancement")
criteria.setAdvancement("triumph:advancements/techtree/refractory_kiln")

alwaysVisible()
