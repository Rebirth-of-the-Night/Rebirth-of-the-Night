setIcon(<craftingcraft:portable_crafting_table>)
setTitle("Crafting Mat")
setDescription("Craft a portable crafting surface to access wooden planks and a myriad recipes on the fly. Craft with tools lying around and a mat, like woven cotton or wool.")


addParent("triumph:advancements/techtree/crafting_table")

//Set the toast notification in the top right corner of the screen to not show up when this advancement is achieved.

criteria.addItem("craftingcraft:portable_crafting_table")

//This function grants the crafting table advancement, skipping it and unlocking further entries. Also gives the craftingTable gamestage
addRewardFunction("triumph:crafted_crafting_mat")
