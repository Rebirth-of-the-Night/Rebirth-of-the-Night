setIcon(<horsepower:grindstone>)
setTitle("Horse Power")
setDescription("Craft a specially modified Horse Grindstone or Horse Press, powered by horses to automate tiring operations.")


addParent("triumph:advancements/techtree/millstone")

//Set the toast notification in the top right corner of the screen to not show up when this advancement is achieved.
setShowToast(true)

//Set the advancement to not send a chat message when it is achieved.
setAnnounceToChat(true)

setRequirements("any")

criteria = addCriteria("Horse Grindstone", "minecraft:inventory_changed")
criteria.addItem("horsepower:grindstone")

criteria = addCriteria("Horse Press", "minecraft:inventory_changed")
criteria.addItem("exsartagine:kettle")

//This function grants the crafting table advancement, skipping it and unlocking further entries. Also gives the craftingTable gamestage
addRewardFunction("triumph:crafted_crafting_mat")

setFrameType("GOAL")

alwaysVisible()
setPos(-168,198)
