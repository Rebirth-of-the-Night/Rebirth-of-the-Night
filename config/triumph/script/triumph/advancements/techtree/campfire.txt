setIcon(<pyrotech:campfire>)
setTitle("Campfire")
setDescription("Place down dried fibers on the ground, add any logs, and light them up with a hand drill to create a campfire for cooking food.")


addParent("triumph:advancements/techtree/dried_fiber")

//Set the toast notification in the top right corner of the screen to not show up when this advancement is achieved.
setShowToast(true)

//Set the advancement to not send a chat message when it is achieved.
setAnnounceToChat(true)

setRequirements("any")

criteria = addCriteria("Light Campfire Drill", "triumph:right_click_block")
criteria.setBlock(<pyrotech:campfire>)
criteria.setItem(<pyrotech:matchstick>)

criteria = addCriteria("Light Campfire F&S", "triumph:right_click_block")
criteria.setBlock(<pyrotech:campfire>)
criteria.setItem(<minecraft:flint_and_steel>)

criteria = addCriteria("Light Campfire Charge", "triumph:right_click_block")
criteria.setBlock(<pyrotech:campfire>)
criteria.setItem(<minecraft:fire_charge>)

//This function grants the lightedCampfire gamestage to hook to. Also grants dried fiber advancement retroactively.
addRewardFunction("triumph:lighted_campfire")
addRewardFunction("triumph:dried_fiber")

alwaysVisible()
setPos(-224,2)
